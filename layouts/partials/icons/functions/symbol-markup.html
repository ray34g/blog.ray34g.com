{{/* layouts/partials/icons/functions/symbol-markup.html input: dict "symbol" "icon-fas-house" "svg" "" returns: string (...) */}} {{- $symbol := .symbol | default "" -}} {{- $svg := .svg | default "" -}} {{- if or (not $symbol) (not $svg) -}}{{- return -}}{{- end -}} {{- $viewBox := "" -}} {{- $matched := findRE `viewBox="([^"]+)"` $svg 1 -}} {{- if gt (len $matched) 0 -}} {{- $viewBox = replaceRE `.*viewBox="([^"]+)".*` `$1` (index $matched 0) -}} {{- end -}} {{- $inner := $svg | replaceRE `(?s)^.*?]*>` "" | replaceRE `(?s).*$` "" | replaceRE `(?s)` "" -}} {{- $symbolOpen := print `` -}} {{- return (print $symbolOpen $inner ``) -}}
