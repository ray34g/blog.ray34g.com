{{- /* Language switcher: - If <= 3 languages: show segmented toggle (always visible) - Else: fallback to dropdown Requirements: - default lang (ja) may omit /ja/ in URL (handled by RelPermalink) - show non-current languages as links when available - when translation missing: disable + tooltip + optional warnf - 404: link to each language home */ -}} {{- $currentLang := .Language.Lang -}} {{- $is404 := eq .Kind "404" -}} {{- $sites := hugo.Sites -}} {{- $numLang := len $sites -}} {{- $items := slice -}} {{- range $sites -}} {{- $targetSite := . -}} {{- $lang := $targetSite.Language.Lang -}} {{- $isCurrent := eq $lang $currentLang -}} {{- $targetHomePrefix := $targetSite.Home.RelPermalink -}} {{- /* この言語の翻訳ページ（存在すれば）を拾う */ -}} {{- $targetPage := false -}} {{- if not $is404 -}} {{- range $.AllTranslations -}} {{- if eq .Language.Lang $lang -}} {{- $targetPage = . -}} {{- end -}} {{- end -}} {{- if $isCurrent -}} {{- $targetPage = $ -}} {{- end -}} {{- end -}} {{- $hasTranslation := and $targetPage (ne $targetPage false) -}} {{- /* 未翻訳表示用（ターゲット言語側 params） */ -}} {{- $badgeText := $targetSite.Params.translationNotAvailableBadge | default "N/A" -}} {{- $notAvailableMessage := $targetSite.Params.translationNotAvailableMessage | default "Not available in this language." -}} {{- /* href/disabled/title（B） */ -}} {{- $href := "#" -}} {{- $disabled := false -}} {{- $title := "" -}} {{- if $isCurrent -}} {{- $disabled = true -}} {{- $title = $.Title -}} {{- else if $is404 -}} {{- $href = $targetHomePrefix -}} {{- else if $hasTranslation -}} {{- $href = $targetPage.RelPermalink -}} {{- $title = $targetPage.Title -}} {{- else -}} {{- $disabled = true -}} {{- $title = $notAvailableMessage -}} {{- end -}} {{- /* 表示ラベル：短い言語コードを優先（例: JA/EN）。無ければ Label */ -}} {{- $short := $targetSite.Params.languageShortName | default (upper $lang) -}} {{- $name := $targetSite.Language.Label | default $short -}} {{- $items = $items | append (dict "lang" $lang "isCurrent" $isCurrent "href" $href "disabled" $disabled "title" $title "short" $short "name" $name "badgeText" $badgeText "hasTranslation" $hasTranslation ) -}} {{- end -}} {{- /* UI: segmented toggle */ -}}
